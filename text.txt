-- //////////// DRAG (corrigido: posicionamento direto, sem tween por frame) //////////// --
local function MakeDraggable(topbarobject, object, clamp)
	local dragging = false
	local dragInput, dragStart, startPos

	local function Update(input)
		local delta = input.Position - dragStart
		local newPos = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
		object.Position = newPos

		if clamp then
			local Viewport = workspace.CurrentCamera.ViewportSize
			local AbsPos, AbsSize = object.AbsolutePosition, object.AbsoluteSize
			local ClampedX = math.clamp(AbsPos.X, 0, math.max(Viewport.X - AbsSize.X, 0))
			local ClampedY = math.clamp(AbsPos.Y, 0, math.max(Viewport.Y - AbsSize.Y, 0))
			if ClampedX ~= AbsPos.X or ClampedY ~= AbsPos.Y then
				object.Position = UDim2.new(newPos.X.Scale, newPos.X.Offset + (ClampedX - AbsPos.X), newPos.Y.Scale, newPos.Y.Offset + (ClampedY - AbsPos.Y))
			end
		end
	end

	topbarobject.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = object.Position

			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)

	topbarobject.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			Update(input)
		end
	end)
end

-- //////////// CONFIG / FLAGS //////////// --
local SettingsLib = {
	SaveSettings = true,
	Flags = {}
}

local CONFIG_FOLDER = "SaturnHub/Library/"
local function GetConfigPath()
	return CONFIG_FOLDER .. LocalPlayer.Name .. ".json"
end

local function GetFlag(name, default)
	if name and SettingsLib.Flags[name] ~= nil then
		return SettingsLib.Flags[name]
	end
	return default
end

local SetFlag -- declarado depois de SaveConfig existir

(getgenv()).LoadConfig = function()
	if not (readfile and writefile and isfile and isfolder and makefolder) then
		return warn("Status : Undetected Executor")
	end
	if not isfolder("SaturnHub") then makefolder("SaturnHub") end
	if not isfolder(CONFIG_FOLDER) then makefolder(CONFIG_FOLDER) end

	local Path = GetConfigPath()
	if isfile(Path) then
		local ok, Decoded = pcall(function()
			return HttpService:JSONDecode(readfile(Path))
		end)
		if ok and type(Decoded) == "table" then
			for key, value in pairs(Decoded) do
				SettingsLib[key] = value
			end
		end
	else
		writefile(Path, HttpService:JSONEncode(SettingsLib))
	end
	print("Library Loaded!")
end

(getgenv()).SaveConfig = function()
	if not (readfile and writefile and isfile and isfolder) then
		return warn("Status : Undetected Executor")
	end
	writefile(GetConfigPath(), HttpService:JSONEncode(SettingsLib))
end

SetFlag = function(name, value)
	if not name then return end
	SettingsLib.Flags[name] = value
	if SettingsLib.SaveSettings then
		(getgenv()).SaveConfig()
	end
end

(getgenv()).LoadConfig()

function Update:SaveSettings()
	return SettingsLib.SaveSettings == true
end

local function BuildToggleControl(parent, text, default, callback, flag)
	local state = default or false
	if flag then state = GetFlag(flag, state) end

	local ToggleButton = Instance.new("TextButton")
	ToggleButton.Size = UDim2.new(1, 0, 0, 32)
	ToggleButton.BackgroundColor3 = Theme.Surface
	ToggleButton.AutoButtonColor = false
	ToggleButton.Text = ""
	ToggleButton.Parent = parent
	CreateRounded(ToggleButton, 8)

	local ToggleLabel = Instance.new("TextLabel")
	ToggleLabel.Position = UDim2.new(0, 12, 0, 0)
	ToggleLabel.Size = UDim2.new(1, -62, 1, 0)
	ToggleLabel.BackgroundTransparency = 1
	ToggleLabel.Text = text
	ToggleLabel.TextColor3 = Theme.Text
	ToggleLabel.Font = Enum.Font.GothamSemibold
	ToggleLabel.TextSize = 12
	ToggleLabel.TextXAlignment = Enum.TextXAlignment.Left
	ToggleLabel.Parent = ToggleButton

	local ToggleOuter = Instance.new("Frame")
	ToggleOuter.Size = UDim2.new(0, 34, 0, 18)
	ToggleOuter.AnchorPoint = Vector2.new(1, 0.5)
	ToggleOuter.Position = UDim2.new(1, -12, 0.5, 0)
	ToggleOuter.BackgroundColor3 = state and Theme.Text or Theme.Off
	ToggleOuter.Parent = ToggleButton
	CreateRounded(ToggleOuter, 99)

	local ToggleInner = Instance.new("Frame")
	ToggleInner.Size = UDim2.new(0, 12, 0, 12)
	ToggleInner.AnchorPoint = Vector2.new(0, 0.5)
	ToggleInner.Position = state and UDim2.new(0, 19, 0.5, 0) or UDim2.new(0, 3, 0.5, 0)
	ToggleInner.BackgroundColor3 = state and Theme.Background or Theme.Text
	ToggleInner.Parent = ToggleOuter
	CreateRounded(ToggleInner, 99)

	local function SetState(newState)
		state = newState
		CreateTween(ToggleInner, {Position = state and UDim2.new(0, 19, 0.5, 0) or UDim2.new(0, 3, 0.5, 0)}, 0.15)
		CreateTween(ToggleOuter, {BackgroundColor3 = state and Theme.Text or Theme.Off}, 0.15)
		CreateTween(ToggleInner, {BackgroundColor3 = state and Theme.Background or Theme.Text}, 0.15)
		if flag then SetFlag(flag, state) end
		if callback then pcall(callback, state) end
	end

	ToggleButton.MouseButton1Click:Connect(function()
		SetState(not state)
	end)

	if callback then pcall(callback, state) end

	local API = {}
	function API:Set(newState) SetState(newState) end
	function API:Get() return state end
	return API
end

local function BuildKeybindControl(parent, text, default, callback, flag)
	local saved = flag and GetFlag(flag, nil)
	local currentKey = (saved and Enum.KeyCode[saved]) or default or Enum.KeyCode.RightControl

	local Holder = Instance.new("Frame")
	Holder.Parent = parent
	Holder.BackgroundColor3 = Theme.Surface
	Holder.Size = UDim2.new(1, 0, 0, 36)
	CreateRounded(Holder, 8)

	local Label = Instance.new("TextLabel")
	Label.Parent = Holder
	Label.BackgroundTransparency = 1
	Label.Position = UDim2.new(0, 12, 0, 0)
	Label.Size = UDim2.new(1, -110, 1, 0)
	Label.Font = Enum.Font.GothamSemibold
	Label.Text = text
	Label.TextColor3 = Theme.Text
	Label.TextSize = 12
	Label.TextXAlignment = Enum.TextXAlignment.Left

	local KeyButton = Instance.new("TextButton")
	KeyButton.Parent = Holder
	KeyButton.AnchorPoint = Vector2.new(1, 0.5)
	KeyButton.Position = UDim2.new(1, -8, 0.5, 0)
	KeyButton.Size = UDim2.new(0, 90, 0, 24)
	KeyButton.BackgroundColor3 = Theme.Off
	KeyButton.AutoButtonColor = false
	KeyButton.Font = Enum.Font.GothamBold
	KeyButton.TextSize = 11
	KeyButton.TextColor3 = Theme.Text
	KeyButton.Text = currentKey.Name
	CreateRounded(KeyButton, 8)

	local listening = false
	local conn

	local function StopListening()
		listening = false
		CreateTween(KeyButton, {BackgroundColor3 = Theme.Off}, 0.15)
		CreateTween(KeyButton, {TextColor3 = Theme.Text}, 0.15)
		KeyButton.Text = currentKey.Name
		if conn then conn:Disconnect() conn = nil end
	end

	KeyButton.MouseButton1Click:Connect(function()
		if listening then StopListening() return end
		listening = true
		KeyButton.Text = "..."
		CreateTween(KeyButton, {BackgroundColor3 = Theme.Text}, 0.15)
		CreateTween(KeyButton, {TextColor3 = Theme.Background}, 0.15)

		conn = UserInputService.InputBegan:Connect(function(input)
			if input.UserInputType == Enum.UserInputType.Keyboard then
				currentKey = input.KeyCode
				if flag then SetFlag(flag, currentKey.Name) end
				if callback then pcall(callback, currentKey) end
				StopListening()
			end
		end)
	end)

	if callback then pcall(callback, currentKey) end

	local API = {}
	function API:Get() return currentKey end
	return API
end
